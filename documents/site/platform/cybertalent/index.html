import base64
from scapy.all import *

op_chars = []
strings = str()

pcap_data = rdpcap("ARP+Storm.pcap")

for pkg in pcap_data:
    ip_layer =  pkg.getlayer("ARP")
    op_chars.append(ip_layer.op)

for x in op_chars:
    print(chr(x), end="")
    strings += chr(x)

print()

print(base64.b64decode(strings).decode())
